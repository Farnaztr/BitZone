ماژول‌های آموزشی جامع

NumPy: قدرت محاسبات عددی

آرایه‌های چندبعدی، عملیات برداری و محاسبات ماتریسی با سرعت باورنکردنی

import numpy as np

# ایجاد آرایه NumPy
arr = np.array([1, 2, 3, 4, 5])
matrix = np.zeros((3, 3))

# عملیات برداری سریع
result = arr * 2 + 5
# ضرب ماتریسی
dot_product = np.dot(matrix, matrix.T)

Pandas: غول مدیریت داده‌ها

خواندن، پاکسازی، تبدیل و تحلیل داده‌های ساختاریافته با DataFrameها

import pandas as pd

# ایجاد DataFrame
data = {'نام': ['علی', 'مریم', 'رضا'],
'سن': [25, 30, 35],
'شهر': ['تهران', 'اصفهان', 'مشهد']}
df = pd.DataFrame(data)

# فیلتر و گروه‌بندی
filtered = df[df['سن'] > 28]
grouped = df.groupby('شهر').mean()
